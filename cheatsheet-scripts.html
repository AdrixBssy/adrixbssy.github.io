Windows / Active Directory

Bloc principal demandé + variantes utiles pour l’examen.

Création d’utilisateurs AD depuis CSV + gestion d’erreurs + log + email RH
PowerShellActive DirectoryCSVEmail
Import-Module ActiveDirectory

$CSVPath    = "C:\Scripts\nouveaux_users.csv"
$LogFile    = "C:\Scripts\creation_users.log"
$Password   = ConvertTo-SecureString "P@ssw0rd2025!" -AsPlainText -Force
$Users      = Import-Csv -Path $CSVPath -Delimiter ","

$SuccessCount = 0
$ErrorCount   = 0
$ErrorsList   = @()

# Paramètres email
$From       = "noreply@techsolutions.local"
$To         = "rh@techsolutions.local"
$SMTPServer = "smtp.techsolutions.local"
$SMTPPort   = 25

function Write-Log {
    param(
        [string]$Level,
        [string]$Message
    )

    $Timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $Line = "[$Timestamp] [$Level] $Message"
    Add-Content -Path $LogFile -Value $Line
}

function Remove-Accents {
    param([string]$Text)

    $Normalized = $Text.Normalize([Text.NormalizationForm]::FormD)
    $Result = $Normalized -replace '\p{M}', ''
    return $Result
}

function Normalize-Login {
    param([string]$Value)

    $Value = Remove-Accents $Value
    $Value = $Value.ToLower()
    $Value = $Value -replace "\s+", ""
    $Value = $Value -replace "[^a-z0-9\.-]", ""
    return $Value
}

"=== Début création utilisateurs : $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') ===" | Out-File -FilePath $LogFile -Encoding UTF8

foreach ($User in $Users) {
    $Prenom   = $User.Prenom.Trim()
    $Nom      = $User.Nom.Trim()
    $Service  = $User.Service.Trim()
    $Fonction = $User.Fonction.Trim()

    $SamBase      = "$(Normalize-Login $Prenom).$(Normalize-Login $Nom)"
    $Sam          = $SamBase
    $UPN          = "$Sam@techsolutions.local"
    $DisplayName  = "$Prenom $Nom"
    $OUPath       = "OU=$Service,OU=Utilisateurs,DC=techsolutions,DC=local"

    try {
        # Vérifier OU
        $OUExists = Get-ADOrganizationalUnit -Filter "DistinguishedName -eq '$OUPath'" -ErrorAction SilentlyContinue
        if (-not $OUExists) {
            throw "L'OU $OUPath n'existe pas"
        }

        # Vérifier utilisateur existant
        $ExistingUser = Get-ADUser -Filter "SamAccountName -eq '$Sam'" -ErrorAction SilentlyContinue
        if ($ExistingUser) {
            throw "L'utilisateur $Sam existe déjà"
        }

        # Création utilisateur
        New-ADUser -Name $DisplayName `
                   -GivenName $Prenom `
                   -Surname $Nom `
                   -SamAccountName $Sam `
                   -UserPrincipalName $UPN `
                   -Path $OUPath `
                   -AccountPassword $Password `
                   -Enabled $true `
                   -ChangePasswordAtLogon $true `
                   -Title $Fonction `
                   -Description "Créé automatiquement le $(Get-Date -Format 'dd/MM/yyyy HH:mm')" `
                   -ErrorAction Stop

        Write-Host "[SUCCÈS] $DisplayName ($Sam)" -ForegroundColor Green
        Write-Log -Level "SUCCES" -Message "Utilisateur créé : $DisplayName ($Sam) dans $OUPath"
        $SuccessCount++
    }
    catch {
        $ErrorMessage = $_.Exception.Message
        Write-Host "[ERREUR] $DisplayName ($Sam) : $ErrorMessage" -ForegroundColor Red
        Write-Log -Level "ERREUR" -Message "$DisplayName ($Sam) : $ErrorMessage"
        $ErrorsList += "- $DisplayName ($Sam) : $ErrorMessage"
        $ErrorCount++
    }
}

$Summary = @"
=== RÉSUMÉ ===
Utilisateurs créés : $SuccessCount
Erreurs rencontrées : $ErrorCount
Total traité : $($Users.Count)
"@

Write-Host $Summary -ForegroundColor Cyan
Add-Content -Path $LogFile -Value ""
Add-Content -Path $LogFile -Value $Summary

$Body = @"
Bonjour,

Le script de création d'utilisateurs Active Directory est terminé.

Résumé :
- Utilisateurs créés : $SuccessCount
- Erreurs rencontrées : $ErrorCount
- Total traité : $($Users.Count)

Détail des erreurs :
$($ErrorsList -join "`r`n")

Le fichier de log complet est joint à ce message.

Cordialement,
Script automatisé
"@

try {
    Send-MailMessage -From $From `
                     -To $To `
                     -Subject "Rapport création utilisateurs AD - $(Get-Date -Format 'dd/MM/yyyy HH:mm')" `
                     -Body $Body `
                     -Attachments $LogFile `
                     -SmtpServer $SMTPServer `
                     -Port $SMTPPort `
                     -Encoding UTF8

    Write-Host "Email envoyé avec succès à $To" -ForegroundColor Green
    Write-Log -Level "INFO" -Message "Email récapitulatif envoyé à $To"
}
catch {
    Write-Host "Erreur lors de l'envoi de l'email : $($_.Exception.Message)" -ForegroundColor Red
    Write-Log -Level "ERREUR" -Message "Erreur d'envoi email : $($_.Exception.Message)"
}

Write-Log -Level "INFO" -Message "Fin du script"

Script de sauvegarde tar + cron
Linuxcrontar
#!/bin/bash

SRC="/home"
DEST="/backup"
DATE=$(date +%F_%H-%M)
ARCHIVE="$DEST/home_backup_$DATE.tar.gz"
LOGFILE="/var/log/backup_home.log"

mkdir -p "$DEST"

tar -czf "$ARCHIVE" "$SRC"
if [ $? -eq 0 ]; then
  echo "[$(date '+%F %T')] Sauvegarde OK : $ARCHIVE" >> "$LOGFILE"
else
  echo "[$(date '+%F %T')] ERREUR sauvegarde" >> "$LOGFILE"
fi

# Crontab : tous les jours à 2h00
# 0 2 * * * /usr/local/bin/backup_home.sh
Sauvegarde rsync avec rotation simple
Linuxrsyncrotation
#!/bin/bash

SRC="/srv/data/"
BASE="/backup/rsync"
TODAY="$BASE/current"
YESTERDAY="$BASE/previous"
LOGFILE="/var/log/rsync_backup.log"

mkdir -p "$BASE"

# Rotation
rm -rf "$YESTERDAY"
[ -d "$TODAY" ] && mv "$TODAY" "$YESTERDAY"
mkdir -p "$TODAY"

rsync -a --delete "$SRC" "$TODAY" >> "$LOGFILE" 2>&1
if [ $? -eq 0 ]; then
  echo "[$(date '+%F %T')] Sauvegarde rsync OK" >> "$LOGFILE"
else
  echo "[$(date '+%F %T')] ERREUR rsync" >> "$LOGFILE"
fi

# Crontab : tous les jours à 1h30
# 30 1 * * * /usr/local/bin/backup_rsync.sh
